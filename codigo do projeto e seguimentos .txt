from PIL import Image
import os
import time

PASTA_ENTRADA = "entrada"
PASTA_SAIDA = "saida"

def garantir_pastas():
    os.makedirs(PASTA_ENTRADA, exist_ok=True)
    os.makedirs(PASTA_SAIDA, exist_ok=True)

def mostrar_menu():
    print("\n" + "=" * 40)
    print(" CONVERSOR UNIVERSAL → PDF")
    print("=" * 40)
    print("1 - Converter IMAGEM para PDF")
    print("0 - Sair")
    print()

def listar_imagens():
    arquivos = os.listdir(PASTA_ENTRADA)
    return [f for f in arquivos if f.lower().endswith((".jpg", ".png", ".jpeg"))]

def escolher_imagens(imagens):
    print("\nImagens encontradas:")
    for i, img in enumerate(imagens, start=1):
        print(f"{i} - {img}")

    print("\nComo deseja converter?")
    print("1 - Converter UMA imagem")
    print("2 - Converter TODAS as imagens")
    print("3 - Converter VÁRIAS imagens")

    escolha = input("Escolha uma opção: ")

    if escolha == "1":
        num = input("Digite o número da imagem: ")
        if num.isdigit() and 1 <= int(num) <= len(imagens):
            return [imagens[int(num) - 1]]

    elif escolha == "2":
        return imagens

    elif escolha == "3":
        nums = input("Digite os números separados por vírgula (ex: 1,3): ")
        selecionadas = []
        for n in nums.split(","):
            n = n.strip()
            if n.isdigit() and 1 <= int(n) <= len(imagens):
                selecionadas.append(imagens[int(n) - 1])
        if selecionadas:
            return selecionadas

    print("❌ Escolha inválida.")
    return []

def converter_imagens_para_pdf(lista_imagens):
    total = len(lista_imagens)

    for i, nome in enumerate(lista_imagens, start=1):
        print(f"🔄 Convertendo {i} de {total}: {nome}")
        caminho_img = os.path.join(PASTA_ENTRADA, nome)
        nome_pdf = os.path.splitext(nome)[0] + ".pdf"
        caminho_pdf = os.path.join(PASTA_SAIDA, nome_pdf)

        try:
            img = Image.open(caminho_img)
            img = img.convert("RGB")
            img.save(caminho_pdf, "PDF")
            time.sleep(0.5)  # só para o usuário perceber o progresso
        except Exception as erro:
            print(f"❌ Erro ao converter {nome}: {erro}")

    print(f"\n✅ {total} imagem(ns) convertida(s) com sucesso!")
    print(f"📂 PDFs salvos na pasta: {PASTA_SAIDA}")

def main():
    garantir_pastas()

    while True:
        mostrar_menu()
        opcao = input("Escolha uma opção: ")

        if opcao == "1":
            imagens = listar_imagens()
            if not imagens:
                print("❌ Nenhuma imagem encontrada na pasta 'entrada'.")
                continue

            selecionadas = escolher_imagens(imagens)
            if selecionadas:
                converter_imagens_para_pdf(selecionadas)

        elif opcao == "0":
            print("👋 Saindo do programa...")
            break

        else:
            print("❌ Opção inválida. Tente novamente.")
